  <CodeBlock title="Propriedades do Button">
```typescript
interface ButtonProps {
  /**
   * Conteúdo do botão
   */
  children: React.ReactNode;

  /**
   * Variante de produto do botão
   * @default "insurance"
   */
  variant?: "insurance" | "banking" | "health" | "negative";

  /**
   * Estilo visual do botão
   * @default "primary"
   */
  styles?: "primary" | "secondary" | "ghost";

  /**
   * Tamanho do botão
   * @default "medium"
   */
  size?: "small" | "medium" | "large";

  /**
   * Se verdadeiro, o botão ocupará 100% da largura do contêiner
   * @default false
   */
  fluid?: boolean;

  /**
   * Se verdadeiro, o botão será desabilitado
   * @default false
   */
  disabled?: boolean;

  /**
   * Se verdadeiro, o botão exibirá um indicador de carregamento
   * @default false
   */
  loading?: boolean;

  /**
   * Função chamada quando o botão é clicado
   */
  onClick?: (event: React.MouseEvent<HTMLButtonElement>) => void;

  /**
   * Tipo do botão HTML nativo
   * @default "button"
   */
  type?: "button" | "submit" | "reset";

  /**
   * Propriedades HTML adicionais
   */
  [x: string]: any;
}
```
  </CodeBlock>
